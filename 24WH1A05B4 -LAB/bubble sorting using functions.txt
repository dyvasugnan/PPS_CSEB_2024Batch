void bubbleSort(int a[],int n){
    int i;
 if(n>0) {
       for(i=0;i<n-1;i++) {
         if(a[i]>a[i+1]){
             swap(&a[i],&a[i+1]);
         }
      bubbleSort(a,n-1);}
  }
else{
  return;
}
}
void swap(int *n1,int *n2){
    int temp;
     temp=*n1;
     *n1=*n2;
     *n2=temp;
